>>> f = open("managed_buildouts.cfg", "w")
>>> f.write("""
... [testbuild]
... location=%s
... python=path/to/local/python
... buildout extends = =base.cfg
... buildout parts = +=pineapple
... """ % "buildout_dir")
>>> f.close()
